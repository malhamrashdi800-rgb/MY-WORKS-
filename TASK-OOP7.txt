CLASS Claim
    ATTRIBUTES
        PRIVATE claimNumber : Number
        PRIVATE dateFiled : String
        PRIVATE status : String
        PRIVATE documentationFolder : DocumentationFolder 
    CONSTRUCTOR(claimNumber, dateFiled, status, documentationFolder)
        SET claimNumber
        SET dateFiled
        SET status
        SET documentationFolder
    END CONSTRUCTOR
    METHOD calculatePayout()
        RETURN 0
    END METHOD
END CLASS
CLASS VehicleClaim EXTENDS Claim
    ATTRIBUTES
        PRIVATE repairEstimate : Number
        PRIVATE excess : Number
    CONSTRUCTOR(claimNumber, dateFiled, status, documentationFolder, repairEstimate, excess)
        CALL Claim(claimNumber, dateFiled, status, documentationFolder)
        SET repairEstimate
        SET excess
    END CONSTRUCTOR
    METHOD calculatePayout()
        RETURN repairEstimate - excess
    END METHOD
END CLASS
CLASS MedicalClaim EXTENDS Claim
    ATTRIBUTES
        PRIVATE treatmentCost : Number
        PRIVATE coverageLimit : Number
    CONSTRUCTOR(claimNumber, dateFiled, status, documentationFolder, treatmentCost, coverageLimit)
        CALL Claim(claimNumber, dateFiled, status, documentationFolder)
        SET treatmentCost
        SET coverageLimit
    END CONSTRUCTOR
    METHOD calculatePayout()
        IF treatmentCost > coverageLimit THEN
            RETURN coverageLimit
        ELSE
            RETURN treatmentCost
        END IF
    END METHOD
END CLASS
CLASS PropertyClaim EXTENDS Claim
    ATTRIBUTES
        PRIVATE damageValue : Number
        PRIVATE maintenanceReduction : Number
    CONSTRUCTOR(claimNumber, dateFiled, status, documentationFolder, damageValue, maintenanceReduction)
        CALL Claim(claimNumber, dateFiled, status, documentationFolder)
        SET damageValue
        SET maintenanceReduction
    END CONSTRUCTOR
    METHOD calculatePayout()
        RETURN damageValue - (damageValue * maintenanceReduction / 100)
    END METHOD
END CLASS
CLASS DocumentationFolder
    ATTRIBUTES
        PRIVATE documents : List<String>
    CONSTRUCTOR()
        CREATE empty documents list
    END CONSTRUCTOR
    METHOD addDocument(document)
        ADD document TO documents
    END METHOD
END CLASS
CLASS Policy
    ATTRIBUTES
        PRIVATE policyNumber : Number
        PRIVATE claims : List<Claim>
    CONSTRUCTOR(policyNumber)
        SET policyNumber
        CREATE empty claims list
    END CONSTRUCTOR
    METHOD addClaim(claim)
        ADD claim TO claims
    END METHOD
END CLASS

   END

CLASS RollerCoaster EXTENDS Ride
    ATTRIBUTES
        PRIVATE peopleInLine : Number
        PRIVATE carCapacity : Number
        PRIVATE rideCycleTime : Number
    CONSTRUCTOR(rideName, location, safetyInspectionLog, peopleInLine, carCapacity, rideCycleTime)
        CALL Ride(rideName, location, safetyInspectionLog)
        SET peopleInLine
        SET carCapacity
        SET rideCycleTime
    END CONSTRUCTOR
    METHOD calculateWaitTime()
        RETURN (peopleInLine / carCapacity) * rideCycleTime
    END METHOD
END CLASS
CLASS WaterRide EXTENDS Ride
    ATTRIBUTES
        PRIVATE peopleInLine : Number
        PRIVATE boatCapacity : Number
        PRIVATE rideCycleTime : Number
        PRIVATE loadingTime : Number
    CONSTRUCTOR(rideName, location, safetyInspectionLog, peopleInLine, boatCapacity, rideCycleTime, loadingTime)
        CALL Ride(rideName, location, safetyInspectionLog)
        SET peopleInLine
        SET boatCapacity
        SET rideCycleTime
        SET loadingTime
    END CONSTRUCTOR
    METHOD calculateWaitTime()
        RETURN (peopleInLine / boatCapacity) * (rideCycleTime + loadingTime)
    END METHOD
END CLASS
CLASS VIPAccessRide EXTENDS RollerCoaster
    ATTRIBUTES
        PRIVATE vipGuests : Number
        PRIVATE standardGuests : Number
    CONSTRUCTOR(rideName, location, safetyInspectionLog, peopleInLine, carCapacity, rideCycleTime, vipGuests, standardGuests)
        CALL RollerCoaster(rideName, location, safetyInspectionLog, peopleInLine, carCapacity, rideCycleTime)
        SET vipGuests
        SET standardGuests
    END CONSTRUCTOR
    METHOD calculateVIPWaitTime()
        RETURN (vipGuests / carCapacity) * rideCycleTime
    END METHOD
    METHOD calculateStandardWaitTime()
        RETURN (standardGuests / carCapacity) * rideCycleTime
    END METHOD
END CLASS

CLASS Server
    ATTRIBUTES
        PRIVATE serialNumber : String
        PRIVATE uptime : Number
        PRIVATE hardwareSpecification : HardwareSpecification
    CONSTRUCTOR(serialNumber, uptime, hardwareSpecification)
        SET serialNumber
        SET uptime
        SET hardwareSpecification
    END CONSTRUCTOR
    METHOD calculateLoad()
        RETURN 0
    END METHOD
END CLASS
CLASS WebServer EXTENDS Server
    ATTRIBUTES
        PRIVATE activeConnections : Number
        PRIVATE maximumConnections : Number
    CONSTRUCTOR(serialNumber, uptime, hardwareSpecification, activeConnections, maximumConnections)
        CALL Server(serialNumber, uptime, hardwareSpecification)
        SET activeConnections
        SET maximumConnections
    END CONSTRUCTOR
    METHOD calculateLoad()
        RETURN (activeConnections / maximumConnections) * 100
    END METHOD
END CLASS
CLASS DatabaseServer EXTENDS Server
    ATTRIBUTES
        PRIVATE queryThroughput : Number
        PRIVATE maximumQueriesPerSecond : Number
    CONSTRUCTOR(serialNumber, uptime, hardwareSpecification, queryThroughput, maximumQueriesPerSecond)
        CALL Server(serialNumber, uptime, hardwareSpecification)
        SET queryThroughput
        SET maximumQueriesPerSecond
    END CONSTRUCTOR
    METHOD calculateLoad()
        RETURN (queryThroughput / maximumQueriesPerSecond) * 100
    END METHOD
    METHOD acceptConnection()
        IF calculateLoad() < 100 THEN
            DISPLAY "Connection Accepted"
        ELSE
            DISPLAY "Server Full"
        END IF
    END METHOD
END CLASS
CLASS CacheServer EXTENDS DatabaseServer
    ATTRIBUTES
        PRIVATE cacheHitRatio : Number
    CONSTRUCTOR(serialNumber, uptime, hardwareSpecification, queryThroughput, maximumQueriesPerSecond, cacheHitRatio)
        CALL DatabaseServer(serialNumber, uptime, hardwareSpecification, queryThroughput, maximumQueriesPerSecond)
        SET cacheHitRatio
    END CONSTRUCTOR
    METHOD calculateLoad()
        RETURN CALL DatabaseServer.calculateLoad() * (1 - cacheHitRatio / 100)
    END METHOD
END CLASS

    ATTRIBUTES
        PRIVATE cpu : String
        PRIVATE ram : Number
        PRIVATE storage : Number
    CONSTRUCTOR(cpu, ram, storage)
        SET cpu
        SET ram
        SET storage
    END CONSTRUCTOR
END CLASS

    END METHOD
