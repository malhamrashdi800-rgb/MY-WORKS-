Task 304: Fitness App Workouts

START
BEGIN

CLASS Workout
    ATTRIBUTES
        PRIVATE workoutDate : String
        PRIVATE duration : Number

    CONSTRUCTOR(workoutDate, duration)
        SET this.workoutDate = workoutDate
        SET this.duration = duration
    END CONSTRUCTOR

    ABSTRACT METHOD calculateCalories()
    END METHOD

    METHOD getWorkoutDate()
        RETURN workoutDate
    END METHOD

    METHOD getDuration()
        RETURN duration
    END METHOD
END CLASS

CLASS RunningWorkout EXTENDS Workout
    ATTRIBUTES
        PRIVATE distance : Number
        PRIVATE pace : Number

    CONSTRUCTOR(workoutDate, duration, distance, pace)
        CALL SUPER(workoutDate, duration)
        SET this.distance = distance
        SET this.pace = pace
    END CONSTRUCTOR

    METHOD calculateCalories()
        RETURN distance * pace * 10
    END METHOD
END CLASS

CLASS CyclingWorkout EXTENDS Workout
    ATTRIBUTES
        PRIVATE distance : Number
        PRIVATE resistanceLevel : Number

    CONSTRUCTOR(workoutDate, duration, distance, resistanceLevel)
        CALL SUPER(workoutDate, duration)
        SET this.distance = distance
        SET this.resistanceLevel = resistanceLevel
    END CONSTRUCTOR

    METHOD calculateCalories()
        RETURN distance * resistanceLevel * 8
    END METHOD
END CLASS

CLASS WeightTrainingWorkout EXTENDS Workout
    ATTRIBUTES
        PRIVATE sets : Number
        PRIVATE reps : Number

    CONSTRUCTOR(workoutDate, duration, sets, reps)
        CALL SUPER(workoutDate, duration)
        SET this.sets = sets
        SET this.reps = reps
    END CONSTRUCTOR

    METHOD calculateCalories()
        RETURN sets * reps * 2
    END METHOD
END CLASS

CLASS FitnessLog
    ATTRIBUTES
        PRIVATE workouts : List<Workout>

    CONSTRUCTOR()
        CREATE empty workouts list
    END CONSTRUCTOR

    METHOD addWorkout(workout : Workout)
        ADD workout TO workouts
    END METHOD

    METHOD calculateTotalCalories()
        SET total = 0
        FOR EACH workout IN workouts
            SET total = total + workout.calculateCalories()
        END FOR
        RETURN total
    END METHOD
END CLASS

END
STOP
Task 305: Video Game Characters

START
BEGIN

CLASS Character
    ATTRIBUTES
        PRIVATE characterName : String
        PRIVATE health : Number
        PRIVATE inventory : Inventory

    CONSTRUCTOR(characterName, health, inventory)
        SET this.characterName = characterName
        SET this.health = health
        SET this.inventory = inventory
    END CONSTRUCTOR

    ABSTRACT METHOD attack()
    END METHOD
END CLASS

CLASS Warrior EXTENDS Character
    ATTRIBUTES
        PRIVATE heavyArmor : Boolean
        PRIVATE meleeStrength : Number

    CONSTRUCTOR(characterName, health, inventory, heavyArmor, meleeStrength)
        CALL SUPER(characterName, health, inventory)
        SET this.heavyArmor = heavyArmor
        SET this.meleeStrength = meleeStrength
    END CONSTRUCTOR

    METHOD attack()
        RETURN "Attacks using melee strength"
    END METHOD
END CLASS

CLASS Mage EXTENDS Character
    ATTRIBUTES
        PRIVATE mana : Number

    CONSTRUCTOR(characterName, health, inventory, mana)
        CALL SUPER(characterName, health, inventory)
        SET this.mana = mana
    END CONSTRUCTOR

    METHOD attack()
        IF mana > 0 THEN
            SET mana = mana - 10
            RETURN "Casts a magic spell"
        ELSE
            RETURN "Not enough mana"
        END IF
    END METHOD
END CLASS

CLASS Archer EXTENDS Character
    ATTRIBUTES
        PRIVATE arrows : Number

    CONSTRUCTOR(characterName, health, inventory, arrows)
        CALL SUPER(characterName, health, inventory)
        SET this.arrows = arrows
    END CONSTRUCTOR

    METHOD attack()
        IF arrows > 0 THEN
            SET arrows = arrows - 1
            RETURN "Shoots an arrow"
        ELSE
            RETURN "No arrows remaining"
        END IF
    END METHOD
END CLASS

CLASS Inventory
    ATTRIBUTES
        PRIVATE items : List<String>

    CONSTRUCTOR()
        CREATE empty items list
    END CONSTRUCTOR

    METHOD addItem(item)
        ADD item TO items
    END METHOD
END CLASS

END

Task 306: Utility Billing System

START
BEGIN

CLASS Bill
    ATTRIBUTES
        PRIVATE billingPeriod : String
        PRIVATE dueDate : String
        PRIVATE meterReading : MeterReading

    CONSTRUCTOR(billingPeriod, dueDate, meterReading)
        SET this.billingPeriod = billingPeriod
        SET this.dueDate = dueDate
        SET this.meterReading = meterReading
    END CONSTRUCTOR

    ABSTRACT METHOD calculateAmountOwed()
    END METHOD
END CLASS


CLASS ElectricityBill EXTENDS Bill
    ATTRIBUTES
        PRIVATE unitRate : Number

    CONSTRUCTOR(billingPeriod, dueDate, meterReading, unitRate)
        CALL SUPER(billingPeriod, dueDate, meterReading)
        SET this.unitRate = unitRate
    END CONSTRUCTOR

    METHOD calculateAmountOwed()
        RETURN meterReading.getConsumption() * unitRate
    END METHOD
END CLASS


CLASS WaterBill EXTENDS Bill
    ATTRIBUTES
        PRIVATE flatRate : Number
        PRIVATE standingCharge : Number

    CONSTRUCTOR(billingPeriod, dueDate, meterReading, flatRate, standingCharge)
        CALL SUPER(billingPeriod, dueDate, meterReading)
        SET this.flatRate = flatRate
        SET this.standingCharge = standingCharge
    END CONSTRUCTOR

    METHOD calculateAmountOwed()
        RETURN (meterReading.getConsumption() * flatRate) + standingCharge
    END METHOD
END CLASS


CLASS GasBill EXTENDS Bill
    ATTRIBUTES
        PRIVATE unitRate : Number
        PRIVATE winterSurcharge : Number

    CONSTRUCTOR(billingPeriod, dueDate, meterReading, unitRate, winterSurcharge)
        CALL SUPER(billingPeriod, dueDate, meterReading)
        SET this.unitRate = unitRate
        SET this.winterSurcharge = winterSurcharge
    END CONSTRUCTOR

    METHOD calculateAmountOwed()
        SET total = meterReading.getConsumption() * unitRate

        IF billingPeriod = "Winter" THEN
            SET total = total + winterSurcharge
        END IF

        RETURN total
    END METHOD
END CLASS


CLASS MeterReading
    ATTRIBUTES
        PRIVATE startReading : Number
        PRIVATE endReading : Number

    CONSTRUCTOR(startReading, endReading)
        SET this.startReading = startReading
        SET this.endReading = endReading
    END CONSTRUCTOR

    METHOD getConsumption()
        RETURN endReading - startReading
    END METHOD
END CLASS


CLASS CustomerAccount
    ATTRIBUTES
        PRIVATE accountID : Number
        PRIVATE customerName : String
        PRIVATE bills : List<Bill>

    CONSTRUCTOR(accountID, customerName)
        SET this.accountID = accountID
        SET this.customerName = customerName
        CREATE empty bills list
    END CONSTRUCTOR

    METHOD addBill(bill : Bill)
        ADD bill TO bills
    END METHOD
END CLASS

END
STOP

Task 307: Ride-Sharing Vehicle Tiers

START
  BEGIN

CLASS Ride
    ATTRIBUTES
        PRIVATE pickupLocation : String
        PRIVATE dropOffLocation : String
        PRIVATE distance : Number
        PRIVATE route : Route

    CONSTRUCTOR(pickupLocation, dropOffLocation, distance, route)
        SET this.pickupLocation = pickupLocation
        SET this.dropOffLocation = dropOffLocation
        SET this.distance = distance
        SET this.route = route
    END CONSTRUCTOR

    ABSTRACT METHOD calculateFare()
    END METHOD
END CLASS


CLASS StandardRide EXTENDS Ride
    ATTRIBUTES
        PRIVATE ratePerKilometer : Number

    CONSTRUCTOR(pickupLocation, dropOffLocation, distance, route, ratePerKilometer)
        CALL Ride(pickupLocation, dropOffLocation, distance, route)
        SET this.ratePerKilometer = ratePerKilometer
    END CONSTRUCTOR

    METHOD calculateFare()
        RETURN this.distance * this.ratePerKilometer
    END METHOD
END CLASS


CLASS PremiumRide EXTENDS Ride
    ATTRIBUTES
        PRIVATE ratePerKilometer : Number
        PRIVATE comfortSurcharge : Number

    CONSTRUCTOR(pickupLocation, dropOffLocation, distance, route, ratePerKilometer, comfortSurcharge)
        CALL Ride(pickupLocation, dropOffLocation, distance, route)
        SET this.ratePerKilometer = ratePerKilometer
        SET this.comfortSurcharge = comfortSurcharge
    END CONSTRUCTOR

    METHOD calculateFare()
        RETURN (this.distance * this.ratePerKilometer) + this.comfortSurcharge
    END METHOD
END CLASS


CLASS SharedRide EXTENDS StandardRide
    ATTRIBUTES
        PRIVATE passengerCount : Number

    CONSTRUCTOR(pickupLocation, dropOffLocation, distance, route, ratePerKilometer, passengerCount)
        CALL StandardRide(pickupLocation, dropOffLocation, distance, route, ratePerKilometer)
        SET this.passengerCount = passengerCount
    END CONSTRUCTOR

    METHOD calculateFare()
        RETURN CALL PARENT.calculateFare() / this.passengerCount
    END METHOD
END CLASS


CLASS Waypoint
    ATTRIBUTES
        PRIVATE location : String
        PRIVATE orderNumber : Number

    CONSTRUCTOR(location, orderNumber)
        SET this.location = location
        SET this.orderNumber = orderNumber
    END CONSTRUCTOR
END CLASS


CLASS Route
    ATTRIBUTES
        PRIVATE waypoints : List<Waypoint>

    CONSTRUCTOR()
        CREATE empty waypoints list
        SET this.waypoints = empty list
    END CONSTRUCTOR

    METHOD addWaypoint(waypoint)
        ADD waypoint TO this.waypoints
    END METHOD
END CLASS
END
STOP
