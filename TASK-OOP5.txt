        PRIVATE color : String

        SET this.x = x
        SET this.y = y

CLASS Circle EXTENDS Shape
    ATTRIBUTES
        PRIVATE radius : Number

    CONSTRUCTOR(color, x, y, radius)
        CALL SUPER(color, x, y)
        SET this.radius = radius
    END CONSTRUCTOR

    METHOD getArea()
        RETURN 3.14 * radius * radius
    END METHOD

    METHOD getPerimeter()
        RETURN 2 * 3.14 * radius
    END METHOD
END CLASS

CLASS Rectangle EXTENDS Shape
    ATTRIBUTES
        PRIVATE width : Number
        PRIVATE height : Number

    CONSTRUCTOR(color, x, y, width, height)
        CALL SUPER(color, x, y)
        SET this.width = width
        SET this.height = height
    END CONSTRUCTOR

    METHOD getArea()
        RETURN width * height
    END METHOD

    METHOD getPerimeter()
        RETURN 2 * (width + height)
    END METHOD
END CLASS

CLASS Triangle EXTENDS Shape
    ATTRIBUTES
        PRIVATE side1 : Number
        PRIVATE side2 : Number
        PRIVATE side3 : Number

    CONSTRUCTOR(color, x, y, side1, side2, side3)
        CALL SUPER(color, x, y)
        SET this.side1 = side1
        SET this.side2 = side2
        SET this.side3 = side3
    END CONSTRUCTOR

    METHOD getArea()
        SET s = (side1 + side2 + side3) / 2
        RETURN SQRT(s * (s - side1) * (s - side2) * (s - side3))
    END METHOD

    METHOD getPerimeter()
        RETURN side1 + side2 + side3
    END METHOD
END CLASS

CLASS Canvas
    ATTRIBUTES
        PRIVATE shapes : List<Shape>

    CONSTRUCTOR()
        CREATE empty shapes list
    END CONSTRUCTOR

    METHOD addShape(shape : Shape)
        ADD shape TO shapes
    END METHOD
END CLASS

CLASS SalariedEmployee EXTENDS Employee
    ATTRIBUTES
        PRIVATE monthlySalary : Number

    CONSTRUCTOR(employeeID, employeeName, hireDate, monthlySalary)
        CALL SUPER(employeeID, employeeName, hireDate)
        SET this.monthlySalary = monthlySalary
    END CONSTRUCTOR

    METHOD calculatePay()
        RETURN monthlySalary
    END METHOD

CLASS HourlyEmployee EXTENDS Employee
    ATTRIBUTES
        PRIVATE hourlyRate : Number
        PRIVATE hoursWorked : Number

    CONSTRUCTOR(employeeID, employeeName, hireDate, hourlyRate, hoursWorked)
        CALL SUPER(employeeID, employeeName, hireDate)

        IF hourlyRate > 0 THEN
            SET this.hourlyRate = hourlyRate
        ELSE
            SET this.hourlyRate = 0
        END IF

        IF hoursWorked >= 0 THEN
            SET this.hoursWorked = hoursWorked
        ELSE
            SET this.hoursWorked = 0
        END IF
    END CONSTRUCTOR

    METHOD calculatePay()
        RETURN hourlyRate * hoursWorked
    END METHOD
END CLASS

CLASS CommissionEmployee EXTENDS Employee
    ATTRIBUTES
        PRIVATE basePay : Number
        PRIVATE sales : Number
        PRIVATE commissionRate : Number

    CONSTRUCTOR(employeeID, employeeName, hireDate, basePay, sales, commissionRate)
        CALL SUPER(employeeID, employeeName, hireDate)
        SET this.basePay = basePay
        SET this.sales = sales
        SET this.commissionRate = commissionRate
    END CONSTRUCTOR

    METHOD calculatePay()
        RETURN basePay + (sales * commissionRate)
    END METHOD
END CLASS

CLASS Payroll
    ATTRIBUTES
        PRIVATE employees : List<Employee>

    CONSTRUCTOR()
        CREATE empty employees list
    END CONSTRUCTOR

    METHOD addEmployee(employee : Employee)
        ADD employee TO employees
    END METHOD

    METHOD calculateTotalPayroll()
        SET total = 0
        FOR EACH employee IN employees
            SET total = total + employee.calculatePay()
        END FOR
        RETURN total
    END METHOD
END CLASS

            SET status = "Completed"

    METHOD validateCard()
        IF LENGTH(cardNumber) = 16 THEN
            RETURN TRUE
        ELSE
            RETURN FALSE
        END IF
    END METHOD

    ATTRIBUTES
        PRIVATE accountNumber : String

    CONSTRUCTOR(amount, accountNumber)
        CALL SUPER(amount)
        SET this.accountNumber = accountNumber
    END CONSTRUCTOR

    METHOD processPayment()
        IF validateAccount() THEN
            DISPLAY "Processing transfer..."
            SET status = "Completed"
        ELSE
            DISPLAY "Invalid account number"
        END IF
    END METHOD

    METHOD validateAccount()
        IF LENGTH(accountNumber) >= 8 THEN
            RETURN TRUE
        ELSE
            RETURN FALSE
        END IF
    END METHOD

    ATTRIBUTES
        PRIVATE walletBalance : Number

    CONSTRUCTOR(amount, walletBalance)
        CALL SUPER(amount)
        SET this.walletBalance = walletBalance

CLASS Card
    ATTRIBUTES
        PRIVATE cardNumber : String
        PRIVATE expiryDate : String
        PRIVATE status : String
        PRIVATE chip : Chip

    CONSTRUCTOR(cardNumber, expiryDate, chip)
        SET this.cardNumber = cardNumber
        SET this.expiryDate = expiryDate
        SET this.status = "Active"
        SET this.chip = chip
    END CONSTRUCTOR

    ABSTRACT METHOD authorisePurchase(amount)
    END METHOD

    METHOD blockCard()
        SET status = "Blocked"
    END METHOD

    METHOD unblockCard()
        SET status = "Active"
    END METHOD
END CLASS

CLASS BankAccount
    ATTRIBUTES
        PRIVATE accountNumber : String
        PRIVATE balance : Number

    CONSTRUCTOR(accountNumber, balance)
        SET this.accountNumber = accountNumber
        SET this.balance = balance
    END CONSTRUCTOR

    METHOD getBalance()
        RETURN balance
    END METHOD

    METHOD deduct(amount)
        SET balance = balance - amount
    END METHOD
END CLASS

CLASS DebitCard EXTENDS Card
    ATTRIBUTES
        PRIVATE account : BankAccount

    CONSTRUCTOR(cardNumber, expiryDate, chip, account)
        CALL SUPER(cardNumber, expiryDate, chip)
        SET this.account = account
    END CONSTRUCTOR

    METHOD authorisePurchase(amount)
        IF account.getBalance() >= amount THEN
            CALL account.deduct(amount)
            RETURN TRUE
        ELSE
            RETURN FALSE
        END IF
    END METHOD
END CLASS

CLASS CreditCard EXTENDS Card
    ATTRIBUTES
        PRIVATE creditLimit : Number
        PRIVATE unpaidPurchases : List<Number>

    CONSTRUCTOR(cardNumber, expiryDate, chip, creditLimit)
        CALL SUPER(cardNumber, expiryDate, chip)
        SET this.creditLimit = creditLimit
        CREATE empty unpaidPurchases list
    END CONSTRUCTOR

    METHOD authorisePurchase(amount)
        IF getOutstandingBalance() + amount <= creditLimit THEN
            ADD amount TO unpaidPurchases
            RETURN TRUE
        ELSE
            RETURN FALSE
        END IF
    END METHOD

    METHOD getOutstandingBalance()
        SET total = 0
        FOR EACH purchase IN unpaidPurchases
            SET total = total + purchase
        END FOR
        RETURN total
    END METHOD

    METHOD makePayment(amount)
        IF amount <= getOutstandingBalance() THEN
            REMOVE amount FROM unpaidPurchases
        END IF
    END METHOD
END CLASS

CLASS Chip
    ATTRIBUTES
        PRIVATE encryptedSecurityData : String

    CONSTRUCTOR(encryptedSecurityData)
        SET this.encryptedSecurityData = encryptedSecurityData
    END CONSTRUCTOR
END CLASS

END

        SET this.isRecording = FALSE
