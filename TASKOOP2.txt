            RETURN TRUE
        END IF

    RETURN FALSE

PUBLIC CALCULATE_AVERAGE()

    SET total TO 0
    SET weightTotal TO 0

    REPEAT FROM I = 1 TO SIZE OF grades

        SET total TO total + (grades[I].GET_SCORE() * grades[I].GET_WEIGHTING())
        SET weightTotal TO weightTotal + grades[I].GET_WEIGHTING()

    END REPEAT

    IF weightTotal = 0 THEN
        RETURN 0
    END IF

    RETURN total / weightTotal

END METHOD

    SET policyID TO policyID
    SET issueDate TO issueDate
    SET policyHolder TO policyHolder
    SET coverages TO LIST []
    SET claims TO LIST []
END CONSTRUCTOR

METHODS

    ADD coverage TO coverages

PUBLIC FILE_CLAIM(claim : Claim)

    IF claim.GET_AMOUNT() <= claim.GET_COVERAGE().GET_LIMIT() THEN

        ADD claim TO claims
        DISPLAY "Claim Filed Successfully"

    ELSE

        DISPLAY "Claim exceeds coverage limit"

    END IF

END METHOD

PUBLIC CALCULATE_REMAINING_COVERAGE()

    SET totalCoverage TO 0
    SET totalClaims TO 0

    REPEAT FROM I = 1 TO SIZE OF coverages

        SET totalCoverage TO totalCoverage + coverages[I].GET_LIMIT()

    END REPEAT

    REPEAT FROM I = 1 TO SIZE OF claims

        IF claims[I].GET_STATUS() EQUALS "Approved" OR
           claims[I].GET_STATUS() EQUALS "Paid" THEN

            SET totalClaims TO totalClaims + claims[I].GET_AMOUNT()

        END IF

    END REPEAT

    RETURN totalCoverage - totalClaims

END METHOD

PUBLIC GET_CLAIMS()
    RETURN claims
END METHOD

END CLASS

        PRINT "Concert Added"

    ELSE

        PRINT "Venue already booked on this date"

    END IF

END METHOD

METHOD removeConcert(concert)
    REMOVE concert FROM concerts
END METHOD

METHOD checkDateAvailable(date)

    FOR EACH concert IN concerts

        IF concert.getConcertDate() = date THEN
            RETURN FALSE
        END IF

    END FOR

    RETURN TRUE

END METHOD
